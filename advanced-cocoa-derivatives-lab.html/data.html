Quick Start Example

import requests

# Fetch historical cocoa prices
api_key = "your_api_key_here"
url = "https://api.cocoa-lab.gh/v1/prices/historical"
params = {
    "symbol": "COCOA",
    "start_date": "2024-01-01",
    "end_date": "2024-12-31",
    "api_key": api_key
}

response = requests.get(url, params=params)
data = response.json()

print(f"Fetched {len(data)} price records")
